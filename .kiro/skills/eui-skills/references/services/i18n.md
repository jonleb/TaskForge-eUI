# i18n / Translation Guide

Internationalization in eUI using ngx-translate.

## Quick Setup

### 1. Configuration (`config/global.ts`)

```typescript
import { GlobalConfig } from '@eui/core';

export const GLOBAL: GlobalConfig = {
  appTitle: 'My App',
  i18n: {
    i18nService: {
      languages: ['en', 'fr', 'de'],
      defaultLanguage: 'en',
    },
    i18nLoader: {
      i18nFolders: ['i18n-eui', 'i18n'], // loads from assets/i18n-eui/ and assets/i18n/
    },
  },
};
```

### 2. Initialize in AppStarterService

```typescript
import { I18nService } from '@eui/core';

@Injectable({ providedIn: 'root' })
export class AppStarterService {
  constructor(
    private i18nService: I18nService,
    private userService: UserService
  ) {}

  start(): Observable<any> {
    return this.userService.getState().pipe(
      take(1),
      switchMap((user) => {
        const lang = user.preferences?.lang;
        return lang 
          ? this.i18nService.init({ activeLang: lang })
          : this.i18nService.init();
      })
    );
  }
}
```

### 3. Translation Files (`assets/i18n/en.json`)

```json
{
  "app.title": "My Application",
  "user.greeting": "Hello, {{name}}!",
  "button.save": "Save",
  "button.cancel": "Cancel"
}
```

## Usage in Templates

### Pipe (recommended)

```html
<h1>{{ 'app.title' | translate }}</h1>
<p>{{ 'user.greeting' | translate:{ name: userName } }}</p>
```

### Directive

```html
<div translate="app.title"></div>
<div [translate]="'user.greeting'" [translateParams]="{ name: userName }"></div>
```

## Usage in Components

```typescript
import { TranslateService } from '@ngx-translate/core';

@Component({...})
export class MyComponent {
  constructor(private translate: TranslateService) {}

  getTranslation() {
    // Reactive (recommended)
    this.translate.getStreamOnTranslationChange('app.title')
      .subscribe(text => console.log(text));

    // Instant (doesn't react to language changes)
    const text = this.translate.instant('app.title');
  }
}
```

## Language Switching

Language changes from App Shell are handled automatically. For programmatic changes:

```typescript
import { I18nService } from '@eui/core';

this.i18nService.updateState({ activeLang: 'fr' });
```

Listen to language changes:

```typescript
this.i18nService.getState().subscribe(state => {
  console.log('Current language:', state.activeLang);
});
```

## Custom Language Labels

```typescript
i18nService: {
  languages: [
    'en',
    'fr',
    { code: 'de', label: 'Deutsch' },
    { code: 'es', label: 'Español' },
  ],
  defaultLanguage: 'en',
}
```

## Loading from API

```typescript
i18nLoader: {
  i18nFolders: ['i18n-eui'],
  i18nServices: ['api/translations/'], // GET api/translations/en
}

// Or with query param
i18nServices: ['api/translations?lang='], // GET api/translations?lang=en
```

## Lazy-Loaded Modules

### 1. Configure module translations (`config/modules.ts`)

```typescript
export const MODULES = {
  portfolio: {
    i18nFolders: ['i18n-portfolio'],
  },
};
```

### 2. Link module

```typescript
import { CoreModule } from '@eui/core';

@NgModule({
  imports: [CoreModule.forChild('portfolio')],
})
export class PortfolioModule {}
```

### 3. Wait for translations

```typescript
import { I18nService, MODULE_NAME_TOKEN } from '@eui/core';

constructor(
  private i18n: I18nService,
  @Inject(MODULE_NAME_TOKEN) private moduleName: string
) {}

ngOnInit() {
  this.i18n.onReady(this.moduleName).subscribe(() => {
    // Translations loaded, safe to use
  });
}
```

## Language Detection Priority

1. Value passed to `init({ activeLang: 'fr' })`
2. Browser language (if in `languages` array)
3. `defaultLanguage` from config
4. Falls back to `'en'`

## Key Interfaces

```typescript
interface I18nServiceConfig {
  languages?: (string | EuiLanguage)[];
  defaultLanguage?: string;
}

interface EuiLanguage {
  code: string;  // ISO 639-1 code
  label: string; // Display name
}
```
